    }

    //Resize if needed

    // If it's the smallest sized symbol, add at the end, otherwise it will be bigger than some element 
    // and index of s will be changed within the for loop
    int index_of_s = st->count;
    //Order the table in correct order, big sizes first
    for (int i = 0; i < st->count; i++)
    {
        if (size_of_type(s->type) >= size_of_type(st->symbols[i]->type))
        {
            index_of_s = i;
            // Add one to all the consecutive symbols' index
            // and shift them one down
            for (int j = st->count; j > i; j--)
            {
                st->symbols[j] = st->symbols[j - 1];
                st->symbols[j]->index++;

            }

            break;
        }
    }
    
    // Add the symbol in its correct spot within the table
    st->count++;
    s->index = index_of_s;
    st->symbols[s->index] = s;

    
    // Finally, recompute the offsets for the symbols with the reorganized table
    // only need to do so for the new inserted symbol and after
    for (int i = s->index; i < st->count; i++)
    {
        st->symbols[i]->offset =  (i == 0) ? 0 : st->symbols[i - 1]->offset + size_of_type(st->symbols[i-1]->type);
        // Align to an offset that is multiple of the variable size
        st->symbols[i]->offset = align_to(st->symbols[i]->offset, size_of_type(st->symbols[i]->type));
    }
    

    //print_symbolTable(st);

    // Add the size of the symbol to the total weight of the table
    st->weight += size_of_type(s->type);


    // Keep track of the padding 
    st->padding = 0;
    for (int i = 1; i < st->count; i++)
    {
        st->padding += st->symbols[i]->offset - (st->symbols[i - 1]->offset + size_of_type(st->symbols[i - 1]->type));

25. Comparison between 8-bit variable and value loaded from memory causes operand-size mismatch
Issue:
When comparing an 8-bit local variable (e.g. loop counter i) with a value loaded from a struct field (a.count), the load from memory is emitted as mov r64, [mem] (64-bit), while the local lives in an 8-bit register (al, bl, etc.). This results in cmp al, rbx — illegal operand sizes — and assembler errors.
This only surfaces when one operand comes from memory (field access, array element) and the other is a known small local/parameter.
Solution:
Before emitting any binary operation (<, >, ==, +, etc.), ensure both operands are in registers of the same size by demoting the larger one to match the smaller one:

if (left.linkedToRegister->size != right.linkedToRegister->size)
{
    int sub = 2;
    while (left.linkedToRegister->size  > right.linkedToRegister->size)
        left.linkedToRegister = left.linkedToRegister->subRegisters[sub--];
    while (right.linkedToRegister->size > left.linkedToRegister->size)
        right.linkedToRegister = right.linkedToRegister->subRegisters[sub--];
}
